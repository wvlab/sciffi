\begin{sciffi}{python}
    from pathlib import Path
    from itertools import islice

    template = r"""
    \begin{frame}[plain]
        \centering
        \includegraphics[width=\paperwidth,height=\paperheight]{
            %s
        }
    \end{frame}
    \clearpage
    """
    
    print("\n".join(
        map(
            lambda x: template % str(x),
            sorted(
                Path("frames").glob("*.png")
            )
        )
    ))
\end{sciffi}
